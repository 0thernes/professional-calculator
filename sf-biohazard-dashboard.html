-- Peak Danger Hours Query
SELECT EXTRACT(HOUR FROM incident_time) AS hour, incident_type, COUNT(*) AS incident_count, ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER ( PARTITION BY incident_type ), 2) AS pct_of_type FROM fact_incidents f JOIN dim_incident_type t ON f.type_id = t.type_id WHERE t.type_name IN ('HUMAN_WASTE', 'NEEDLE') GROUP BY 1, 2 ORDER BY hour, incident_type;
-- Neighborhood Risk Score with Window Functions
WITH risk_calc AS ( SELECT n.name AS neighborhood, COUNT(CASE WHEN t.type_name = 'HUMAN_WASTE' THEN 1 END) AS hw_count, COUNT(CASE WHEN t.type_name = 'NEEDLE' THEN 1 END) AS needle_count, COUNT(*) AS total, ROUND(LN(COUNT(*) + 1) * 10, 2) AS risk_score FROM fact_incidents f JOIN dim_neighborhood n ON f.neighborhood_id = n.neighborhood_id JOIN dim_incident_type t ON f.type_id = t.type_id GROUP BY n.name ) SELECT *, RANK() OVER(ORDER BY risk_score DESC) AS rank FROM risk_calc ORDER BY risk_score DESC;
-- Year-over-Year with LAG
SELECT d.year, t.type_name, COUNT(*) AS total, COUNT(*) - LAG(COUNT(*)) OVER ( PARTITION BY t.type_name ORDER BY d.year ) AS yoy_change, ROUND( (COUNT(*) - LAG(COUNT(*)) OVER ( PARTITION BY t.type_name ORDER BY d.year )) * 100.0 / NULLIF(LAG(COUNT(*)) OVER ( PARTITION BY t.type_name ORDER BY d.year ), 0), 2 ) AS yoy_pct FROM fact_incidents f JOIN dim_date d ON f.date_id = d.date_id JOIN dim_incident_type t ON f.type_id = t.type_id GROUP BY d.year, t.type_name ORDER BY d.year, t.type_name;
